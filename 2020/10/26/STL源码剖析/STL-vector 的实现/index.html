类似于vector的数据结构实现

类似于vector的数据结构实现

预期实现的功能

  • 可以对内置类型以及自定义数据类型进行存储
  • 将数组中的数据存储到堆区
  • 构造函数
    • 空构造
    • 传入容量
    • 传入容量和初始化值
  • 提供对应的拷贝构造函数和拷贝赋值函数防止浅拷贝
  • 提供尾插法核尾删法
  • 可以通过下标访问数组元素
  • 可以获取当前的容量和大小

代码

template <typename _Tp>
class MyArray{
    
    typedef _Tp ValueType;
    
private:
    
    int _size; //大小
    int _capacity; //容量
    ValueType *address; //管理地址
    
    //实际上vector是管理的三个迭代器
public:

    // 构造函数
    MyArray(){
        //全部初始化为0
        _size  = 0;
        _capacity = 0;
        address = nullptr;
    }
    MyArray(int capacity){
        //容量设置好以及分配好对应的堆内存
        _size = 0;
        _capacity = capacity;
        address = new ValueType[capacity];
    }
    MyArray(int capacity,ValueType val){
        //容量和大小都设置好,以及分配对应的堆内存并进行初始化
        _size = capacity;
        _capacity = capacity;
        address = new ValueType[capacity];
        for (int i = 0; i < capacity; ++i) {
            *(address + i) = val;
        }
    }
    //拷贝构造
    MyArray(const MyArray &arr) {
        this->_size = arr._size;
        this->_capacity = arr._capacity;
        this->address = new ValueType[this->_capacity]; //深拷贝
    }
    //拷贝赋值
    MyArray & operator=(const MyArray & arr){
        //检查原本是否存在管理的内存
        if(this->address != nullptr){
            this->_capacity = 0;
            this->_size = 0;
            delete [] this->address;
            this->address = nullptr;
        }
        this->_size = arr._size;
        this->_capacity = arr._capacity;
        this->address = new ValueType[this->_capacity]; //深拷贝
        for (int i = 0; i < this->_size; ++i) {
            this->address[i] = arr.address[i];
        }
    }
    //析构函数
    ~MyArray() {
        if(address != nullptr) {
            delete[] address;
            this->address = nullptr;
        }
    }

    //取值
    int size(){
        return _size;
    }
    int capacity(){
        return _capacity;
    }

    //[]重载
    ValueType & operator[](int n){ //可以作为左值
        return this->address[n];
    }

    void adjust_capacity(const ValueType & obj){ //内存扩容
        if(_capacity == _size){
            //申请2倍内存释放原本内存进行拷贝
            std::cout << "调整" << std::endl;
            _capacity = _capacity == 0 ? 1 : 2 * _capacity;
            ValueType * new_address = new ValueType[_capacity];
            // 放入当前尾部插入的元素 还没有考虑 vector.insert()
            new_address[_size] = obj;
            _size ++;
            delete [] address;
            address = new_address;
        }
        return;
    }

    //尾插尾删
    void push_back(const ValueType & obj){
        if(_capacity == _size){
            adjust_capacity(obj);
            return;
        }
        this->address[_size] = obj;
        this->_size++;
    }
    void pop_back(){
        if(_size == 0){
            return;
        }
        this->_size --;
    }
};
