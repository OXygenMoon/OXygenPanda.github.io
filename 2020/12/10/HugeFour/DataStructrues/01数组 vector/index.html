数组

数组的特性较为简单, 在内存空间中连续, 存储同一类型元素, 查询时间复杂度为O(1), 插入删除时间复杂度为O(n). 数据与数据的关系只有先后关系.

// linear.c

#include "linear.h"

template <typename _Tp>
class myvector {
protected:
	typedef _Tp	           value_type;
	typedef value_type*    pointer;
	typedef pointer        iterator;
	typedef _Tp&           reference;

private:
	iterator start;
	iterator finish;
	iterator end_of_storage;

public:
	// Construction & Deconstruction
	myvector() { start = finish = end_of_storage = nullptr; }
	myvector(int capacity, value_type initElem = 0) { if(capacity > 0) initalloc(capacity,initElem); }
	myvector(const myvector& LeftVector) {
		int size = LeftVector.size();
		this->resize(size);
		for(int i = 0; i < size ; ++ i){
			this[i] = LeftVector[i];
		}
	}
//	myvector(myvector && RightVector) noexcept {}
	~myvector() {
		this->clear();
		start = finish = end_of_storage = nullptr;
	};

	// Operations
	myvector & operator=(const myvector & LeftVector) {
		if(&LeftVector == this)
			return *this;
		if(start != 0){
			this->clear();
		}
		int size = LeftVector.size();
		this->resize(size);
		for(int i = 0; i < size ; ++ i){
			this[i] = LeftVector[i];
		}
		return *this;
	}
//	myvector & operator=(myvector && RightVector) noexcept {}
	bool operator==(const myvector & LeftVector) {
		if(&LeftVector == this)	return true;
		if(this->size() != LeftVector.size())	return false;
		for(int i = 0 ; i < this->size() ; ++ i){
			if(this[i] != LeftVector[i])
				return false;
		}
		return true;
	}
	bool operator!=(const myvector & LeftVector) {
		return !operator==(LeftVector);
	}

	// Capacity
	int size() {
		return (finish - start);
	}
	int capacity() {
		return (end_of_storage - start);
	}
	bool empty() {
		return (this->size() == 0);
	}

	// Visit
	value_type front() {
		//empty应该抛出异常
		return *begin();
	}
	value_type back() {
		//empty应该抛出异常
		return *(end() - 1);
	}
	iterator begin() {
		return start;
	}
	iterator end() {
		return finish;
	}

	// Modifiers
	void clear() {}
	void insert(iterator location, value_type InsertElem) {}
	void erase(iterator location) {}
	void erase(iterator _begin, iterator _end) {}
	void push_back(value_type PushElem) {}
	void emplace_back(value_type PushElem) {}
	void pop_back() {}
	void resize() {}



private:
	void initalloc(int capacity, value_type Elem){
		start = new value_type[capacity];
		iterator it = start;
		for(int i = 0 ; i < capacity ; ++ i, ++ it) {
			*it = Elem;
		}
		end_of_storage = it + 1;
		finish = it;
	}
	void growalloc(){

	}
};
